Map/Reduce

Map/Reduce on a cluster:

Greatly simplified single-machine example (linux):

> cat largefile.txt | python mapper.py | python reducer.py > outfile.txt

Mapper

# Executed on many cores on many computers across a network
def mapper():
    results = collections.defaultdict(int)
    # Read the input line by line
    for line in sys.stdin: 
        line = line.split("\t")
        # Custom Event parser
        e = Event(line, filter="dogma::Damaged_Player") 
        # Only take valid events of this type
        if e is None:
            continue
        # Go through each moduleType|chargeType|numShots and sum up the damage (map)
        for l in e.moduleTypeID_chargeTypeID_numShots_damageSum:
            results[(e.solarSystemID, e.shipTypeID, e.attackerShipTypeID, l[0], l[1], l[2])] += int(l[3])

    # Print mapper output for reducer to process further
    for key, val in results.iteritems():
        print "%s\t%s" % ("|".join(key), val)

Reducer

# Executed on 1 machine unless mapper output is sorted so it can be split into multiple reducers
def reducer():
    countsByEvent = collections.defaultdict(int)
    #Read the input line by line
    for line in sys.stdin:
        line = line.strip()
        # From the mapper we know the lines from it are on a key\tvalue format
        c = line.split("\t")
        # Further reduce the map output into final form
        countsByEvent[c[0]] += int(c[1])
    
    keys = countsByEvent.keys()
    keys.sort()
    # Output
    for k in keys:
        k = k.split("|") + countsByEvent[k]
        print("\t".join(k))

Loading dataset into R

Map/Reduce reduced damage, but not number of shots, it’s easy to fix that in R

damage <- read_tsv("data/pvpdamage.tsv",
                   col_names=c("solarSystemID", "shipTypeID", "attackerShipTypeID", "moduleTypeID", "chargeTypeID", "numShots", "damage"),
                   col_types=cols(numShots = col_double(), 
                                  damage = col_double()),
                   na = "None") %>%
  group_by(solarSystemID, shipTypeID, attackerShipTypeID, moduleTypeID, chargeTypeID) %>%
  summarize(numShots = sum(numShots), 
            damage = sum(damage)) %>%
  ungroup()
